# barcode_qr.py
import qrcode
from barcode import Code128
from barcode.writer import ImageWriter
import io
import logging

# Use the same logger as the main application
logger = logging.getLogger('BillboardSplitter')

def generate_barcode_in_memory(barcode_type: str, data: str, show_text: bool = True) -> bytes | None:
    """
    Generates a barcode or QR code as a PNG image and returns its bytes from an in-memory buffer.
    This is the single, optimized function for generating all codes.
    """
    if not data:
        logger.warning("Attempting to generate a barcode for empty data.")
        return None

    try:
        buffer = io.BytesIO()
        
        if barcode_type == 'code_qr':
            # Generate QR code using qrcode library
            qr_img = qrcode.make(data)
            qr_img.save(buffer, format='PNG')
            logger.debug(f"Generated QR code PNG in memory for: {data[:30]}...")

        elif barcode_type == 'code_barcode':
            # Configure the writer for the barcode image
            writer_options = {
                'write_text': show_text,
                'font_size': 14,
                'text_distance': 2.0,
                'module_height': 10.0,
                'quiet_zone': 2.0, # Add some padding
            }
            # Generate Code128 barcode using python-barcode library
            code = Code128(data, writer=ImageWriter(format='PNG'))
            code.write(buffer, options=writer_options)
            logger.debug(f"Generated Code128 PNG in memory for: {data[:30]}...")

        else:
            logger.error(f"Unknown barcode type requested: {barcode_type}")
            return None

        buffer.seek(0)
        return buffer.getvalue()

    except Exception as e:
        logger.error(f"Error generating {barcode_type} for data '{data}': {e}", exc_info=True)
        return None

import time

TEST_FILES = ["billboard1.pdf", "billboard2.pdf", "billboard3.pdf"]

SCENARIOS = {

# --- Base settings common to all scenarios ---
BASE_SETTINGS = {
    "overlap": 20.0,
    "rows": 2,
    "cols": 5,
    "layout": "layout_vertical",
    "enable_reg_marks": True,
    "registration_mark_type": "reg_mark_cross",
    "enable_barcode": True,

    "bryt_order": "bryt_order_1",
    "white_stripe": 10.0,
    "add_white_to_overlap": True,
}

def run_test_scenario(splitter_instance, scenario_name, scenario_config):
    """Runs a single test scenario and prints the results."""
    output_dir = scenario_config["output_dir"]
    test_settings = {**BASE_SETTINGS, **scenario_config["settings"]}

    print(f"--- Running Scenario: {scenario_name} ---")

    if os.path.exists(output_dir):
        shutil.rmtree(output_dir)
    os.makedirs(output_dir)
    print(f"Output directory: '{output_dir}'")
    print("-" * 50)

    total_time = 0
    for test_file in TEST_FILES:
        if not os.path.exists(test_file):
            print(f"Test file '{test_file}' not found. Skipping.")
            continue

        print(f"Processing file: {test_file}...")
        start_time = time.time()

        success = splitter_instance.split(
            input_pdf_path=test_file,
            output_dir=output_dir,
            **test_settings
        )

        end_time = time.time()
        duration = end_time - start_time
        total_time += duration

        if success:
            print(f"Completed in: {duration:.4f} seconds")
        else:
            print(f"An error occurred during processing. Time: {duration:.4f} seconds")

        print("-" * 50)

    print(f"\n--- Scenario '{scenario_name}' Summary ---")
    avg_time = total_time / len(TEST_FILES) if TEST_FILES else 0
    print(f"Total time for {len(TEST_FILES)} files: {total_time:.4f} seconds")
    print(f"Average time per file: {avg_time:.4f} seconds")
    print("-" * 50 + "\n")
    return total_time

def profiled_task():
    """The main function to be profiled, running all test scenarios."""
    print("--- Starting Comprehensive Verification Benchmark ---")

    try:
        config.load_settings()
        logging.getLogger('BillboardSplitter').setLevel(logging.WARNING)
        splitter_instance = splitter.SplitPDF(config)
    except Exception as e:
        print(f"Error during initialization: {e}")
        return

    total_benchmark_time = 0
    for name, scenario_config in SCENARIOS.items():
        total_benchmark_time += run_test_scenario(splitter_instance, name, scenario_config)

    print(f"\n=== Total Benchmark Time for All Scenarios: {total_benchmark_time:.4f} seconds ===")

def run_profiler():
    """Runs cProfile on the main task and saves the results."""
    profiler = cProfile.Profile()
    profiler.enable()

    profiled_task()

    profiler.disable()

    s = io.StringIO()
    ps = pstats.Stats(profiler, stream=s).sort_stats('cumulative')
    ps.print_stats(40)

    profile_results = s.getvalue()

    try:
        with open("profiling_results_final.txt", "w+") as f:
            f.write(profile_results)
        print("\nFinal profiling results saved to 'profiling_results_final.txt'")
    except Exception as e:
        print(f"\nCould not save final profiling results: {e}")

    print("\n--- Final Profiling Results (cProfile) ---")
    print(profile_results)
    print("------------------------------------------")

if __name__ == "__main__":
    run_profiler()

#!/usr/bin/env python3
"""
Bootstrap aplikacji:
  • zakłada/aktualizuje .venv
  • instaluje requirements.txt
  • uruchamia main.py z wyciszoną konsolą
"""

import os
import subprocess
import sys
from pathlib import Path
import venv
import platform

BASE_DIR = Path(__file__).resolve().parent
VENV_DIR = BASE_DIR / ".venv"
REQ_FILE = BASE_DIR / "requirements.txt"
MAIN_FILE = BASE_DIR / "main.py"


def ensure_venv() -> Path:
    """Tworzy .venv, o ile go nie ma, oraz instaluje wymagane paczki."""
    if not VENV_DIR.exists():
        print("Tworzę izolowane środowisko…")
        venv.create(VENV_DIR, with_pip=True)
    # python/pip w zależności od systemu
    py_bin = VENV_DIR / ("Scripts" if os.name == "nt" else "bin")
    python = py_bin / ("python.exe" if os.name == "nt" else "python3")
    #  instalacja/aktualizacja zależności
    if REQ_FILE.exists():
        subprocess.check_call([str(python), "-m", "pip", "install", "--upgrade", "-r", str(REQ_FILE)])
    return python


def launch_app(python: Path) -> None:
    """Uruchamia main.py bez konsoli (Windows) lub w tle (Unix)."""
    kwargs = {
        "cwd": BASE_DIR,
        "stdout": subprocess.DEVNULL,
        "stderr": subprocess.DEVNULL,
        "stdin": subprocess.DEVNULL,
    }

    if os.name == "nt":
        # używamy pythonw.exe, by nie pokazywać konsoli
        pythonw = python.with_name("pythonw.exe")
        subprocess.Popen([str(pythonw), str(MAIN_FILE)], **kwargs, creationflags=subprocess.DETACHED_PROCESS)
    else:
        # Unix: detach (nowe session id) i wyciszenie wyjść
        subprocess.Popen([str(python), str(MAIN_FILE)], **kwargs, start_new_session=True)


def main() -> None:
    if not MAIN_FILE.exists():
        sys.exit("Brak main.py – przerwano.")
    python = ensure_venv()
    launch_app(python)


if __name__ == "__main__":
    main()

def get(key, default=None):
    """
    Umożliwia wywołania w stylu config.get('logger', None) 
    tak jak przy pracy z dict. Zwraza wartość z settings lub default.
    """
    return settings.get(key, default)

def load_settings(filepath="settings.json"):
    global settings

    if getattr(sys, 'frozen', False):
        base_dir = os.path.dirname(sys.executable)
        bundled_dir = sys._MEIPASS
    else:
        base_dir = os.path.dirname(os.path.abspath(__file__))
        bundled_dir = base_dir

    full_path = os.path.join(base_dir, filepath)

    if not os.path.exists(full_path):
        source_path = os.path.join(bundled_dir, filepath)
        if os.path.exists(source_path):
            shutil.copyfile(source_path, full_path)
        else:
            settings = default_settings.copy()
            save_settings(filepath)
            return settings

    try:
        with open(full_path, 'r', encoding='utf-8') as f:
            loaded_settings = json.load(f)
            settings = default_settings.copy()
            settings.update(loaded_settings)
    except Exception:
        settings = default_settings.copy()
        save_settings(filepath)

    if getattr(sys, 'frozen', False):
        settings["default_input_dir"] = os.path.join(base_dir, "input")
        if not settings.get("file_path"):
            sample_input = os.path.join(base_dir, "input", "bilboard3.pdf")
            if os.path.exists(sample_input):
                 settings["file_path"] = sample_input
        
        # --- POCZĄTEK ZMIANY ---
        # Ustaw domyślny katalog wyjściowy tylko, jeśli nie jest już zdefiniowany w ustawieniach.
        if not settings.get("output_dir"):
            settings["output_dir"] = os.path.join(base_dir, "output")
        
        # Ustaw domyślną ścieżkę logów tylko, jeśli nie jest już zdefiniowana.
        if not settings.get("log_file_path"):
            settings["log_file_path"] = os.path.join(base_dir, "logs", "app.log")
        # --- KONIEC ZMIANY ---

    return settings




def save_settings(filepath="settings.json"):
    if getattr(sys, 'frozen', False):
        base_dir = os.path.dirname(sys.executable)
    else:
        base_dir = os.path.dirname(os.path.abspath(__file__))
    full_path = os.path.join(base_dir, filepath)
    with open(full_path, 'w', encoding='utf-8') as f:
        json.dump(settings, f, indent=4, ensure_ascii=False)

def initialize_directories():
    # Dla exe (frozen) używamy katalogu z exe, a nie sys._MEIPASS
    if getattr(sys, 'frozen', False):
        base_dir = os.path.dirname(sys.executable)
        # Katalog z zasobami w trybie frozen (dla pliku przykładowego)
        bundled_dir = sys._MEIPASS
    else:
        # W trybie developerskim używamy katalogu, gdzie jest ten skrypt
        base_dir = os.path.dirname(os.path.abspath(__file__))
        bundled_dir = base_dir # W trybie dev zasoby są tam gdzie kod

    input_dir = os.path.join(base_dir, "input")
    output_dir = os.path.join(base_dir, "output")
    logs_dir = os.path.join(base_dir, "logs")

    # Tworzymy katalogi, jeśli nie istnieją
    for d in [input_dir, output_dir, logs_dir]:
        if not os.path.exists(d):
            try:
                os.makedirs(d, exist_ok=True)
                # Zamiast print:
                logger.info(LANG["directory_created"].format(d))
            except Exception as e:
                # Zamiast print:
                logger.error(LANG["directory_creation_error"].format(d, e))

def load_private_key(path):
    """
    Ładuje prywatny klucz RSA z pliku PEM.
    Jeśli klucz jest zaszyfrowany, należy przekazać hasło (tutaj zakładamy brak hasła).
    """
    with open(path, "rb") as key_file:
        private_key = serialization.load_pem_private_key(
            key_file.read(),
            password=None
        )
    return private_key

def generate_license(hwid, private_key):
    """
    Podpisuje podany HWID przy użyciu klucza prywatnego i zwraca klucz licencyjny zakodowany w Base64.
    """
    signature = private_key.sign(
        hwid.encode('utf-8'),
        padding.PSS(
            mgf=padding.MGF1(hashes.SHA256()),
            salt_length=padding.PSS.MAX_LENGTH
        ),
        hashes.SHA256()
    )
    license_key = base64.b64encode(signature).decode('utf-8')
    return license_key

def activate_license(license_key, logger=None):
    """
    Aktywuje licencję, weryfikując ją i zapisując do konfiguracji.
    
    Args:
        license_key: Klucz licencyjny do aktywacji.
        logger: Opcjonalny logger do zapisywania informacji o błędach.
        
    Returns:
        bool: True jeśli aktywacja się powiodła, False w przeciwnym razie.
        str: Komunikat o błędzie lub None jeśli aktywacja się powiodła.
    """
    if not license_key:
        return False, translate("enter_license_key")
    
    hwid = get_hwid()
    if logger:
        logger.debug(f"{translate('computer_hwid')}: {hwid}")
    
    try:
        public_key = load_public_key()
    except Exception as e:
        if logger:
            logger.error(f"{translate('public_key_load_error_log')}: {e}", exc_info=True)
        return False, translate("public_key_load_error").format(error=e)
    
    try:
        signature = base64.b64decode(license_key)
    except Exception as e:
        if logger:
            logger.error(f"{translate('license_key_decode_error')}: {e}", exc_info=True)
        return False, translate("invalid_license_format").format(error=e)
    
    try:
        public_key.verify(
            signature,
            hwid.encode('utf-8'),
            padding.PSS(
                mgf=padding.MGF1(hashes.SHA256()),
                salt_length=padding.PSS.MAX_LENGTH
            ),
            hashes.SHA256()
        )
        config.settings["license_key"] = license_key
        return True, None
    except Exception as e:
        if logger:
            logger.error(f"{translate('license_activation_error')}: {e}", exc_info=True)
        return False, translate("activation_error").format(error=e)

    def find_key_for_value(self, value_to_find, possible_keys):
        """
        Znajduje klucz językowy w LANG, którego przetłumaczona wartość
        odpowiada podanej wartości (value_to_find).
        Przeszukuje tylko w ramach podanej listy możliwych kluczy.
        """
        global LANG  # Potrzebujemy dostępu do globalnego słownika językowego
        for key in possible_keys:
            if key in LANG and LANG[key] == value_to_find:
                return key
        # Jeśli nie znaleziono dokładnego dopasowania, zwróć pierwszy możliwy klucz jako domyślny
        self.logger.warning(f"Nie znaleziono klucza dla wartości '{value_to_find}'. Używam domyślnego '{possible_keys[0]}'.")
        return possible_keys[0]

    def browse_file(self):
        # --- POCZĄTEK ZMIANY ---
        # Sprawdź, czy w konfiguracji zapisana jest ścieżka do ostatnio używanego pliku
        last_file_path = config.settings.get("file_path")

        # Jeśli ścieżka istnieje i prowadzi do istniejącego pliku,
        # użyj katalogu tego pliku jako lokalizacji początkowej.
       
        if last_file_path and os.path.isfile(last_file_path):
            initial_dir = os.path.dirname(last_file_path)
        else:
            # W przeciwnym razie (np. pierwsze uruchomienie), użyj domyślnego katalogu wejściowego
            initial_dir = config.settings.get("default_input_dir", get_base_dir())
        # --- KONIEC ZMIANY ---

        file_path = filedialog.askopenfilename(
            initialdir=initial_dir,
   
            title=translate("select_file_title"),
            filetypes=[(translate("supported_files"), "*.pdf *.jpg *.jpeg *.tiff *.tif"), (translate("all_files"), "*.*")]
        )
        if file_path:
            self.file_path.set(file_path)
            config.settings["file_path"] = file_path
            
            # --- KLUCZOWA ZMIANA ---
            # Wyczyść listę starych plików wyjściowych, aby podgląd się zaktualizował poprawnie
            self.generated_output_files = []
            self.generated_output_pdf = None
            # --- KONIEC ZMIANY ---

            # Po wybraniu nowego pliku, odśwież oba podglądy.
            if hasattr(self, 'home_tab_instance'):
                self.home_tab_instance.update_preview()
                self.home_tab_instance.update_output_preview()

    def save_profile(self):
        profile_name = self.profile_name_var.get().strip()
        if not profile_name:
            messagebox.showwarning(translate("warning_no_name"), translate("profile_enter_name"))
            return

        if not self.save_settings_local():
            return

        # Utwórz kopię ustawień, usuwając ścieżkę pliku wejściowego
        profile_settings = config.settings.copy()
        profile_settings.pop("file_path", None)
        self.profiles_data[profile_name] = profile_settings
        save_profiles_to_file(self.profiles_data)

        if profile_name in self.profile_list:
            self.profile_list.remove(profile_name)
        self.profile_list.insert(0, profile_name)
        self.profile_list = self.profile_list[:10]

        if self.profile_combo is not None:
            self.profile_combo['values'] = self.profile_list
        else:
            self.logger.warning("profile_combo is None; cannot update its values.")

        if hasattr(self, 'home_tab_instance') and hasattr(self.home_tab_instance, 'recent_profiles_container'):
            self.home_tab_instance._create_recent_profile_buttons(self.home_tab_instance.recent_profiles_container)

        messagebox.showinfo(translate("profile_saved_title"), translate("profile_saved").format(profile=profile_name))
        self.logger.info(translate("logger_profile_saved").format(profile=profile_name))

    def load_profile(self):
        profile_name = self.profile_name_var.get().strip()
        if not profile_name:
            messagebox.showwarning(translate("warning_no_profile"), translate("select_profile"))
            return
        if profile_name not in self.profiles_data:
            messagebox.showerror(translate("error"), translate("profile_not_found").format(profile=profile_name))
            self.logger.warning(translate("logger_profile_not_found").format(profile=profile_name))
            return

        profile_settings = self.profiles_data[profile_name]
        # Zapamiętaj bieżący plik wejściowy i przywróć go po wczytaniu profilu
        current_file_path = config.settings.get("file_path", "")
        config.settings.clear()
        config.settings.update(profile_settings.copy())
        config.settings["file_path"] = current_file_path

        try:
            initialize_directories()  # Inicjalizacja katalogów po zmianie ustawień
        except Exception as e:
            messagebox.showerror(translate("init_error_title"), translate("init_error").format(error=e))
            self.logger.error(translate("logger_init_error").format(error=e), exc_info=True)
            return

        self.update_gui_from_config()
        messagebox.showinfo(translate("profile_loaded_title"), translate("profile_loaded").format(profile=profile_name))
        self.logger.info(translate("logger_profile_loaded").format(profile=profile_name))

    def delete_profile(self):
        profile_name = self.profile_name_var.get().strip()
        if not profile_name:
            messagebox.showwarning(translate("warning_no_profile"), translate("select_profile"))
            return
        if profile_name not in self.profiles_data:
            messagebox.showerror(translate("error"), translate("profile_not_found").format(profile=profile_name))
            self.logger.warning(translate("logger_profile_not_found").format(profile=profile_name))
            return

        if messagebox.askyesno(translate("confirm_delete_title"), translate("confirm_delete_profile").format(profile=profile_name)):
            del self.profiles_data[profile_name]
            save_profiles_to_file(self.profiles_data)

            if profile_name in self.profile_list:
                self.profile_list.remove(profile_name)
                if self.profile_combo is not None:
                    self.profile_combo['values'] = self.profile_list
            self.profile_name_var.set("")

            if hasattr(self, 'home_tab_instance') and hasattr(self.home_tab_instance, 'recent_profiles_container'):
                self.home_tab_instance._create_recent_profile_buttons(self.home_tab_instance.recent_profiles_container)

            messagebox.showinfo(translate("profile_deleted_title"), translate("profile_deleted").format(profile=profile_name))
            self.logger.info(translate("logger_profile_deleted").format(profile=profile_name))

    def save_settings_local(self) -> bool:
        """Pobiera wartości z GUI i aktualizuje globalny słownik config.settings."""
        self.logger.debug(translate("logger_start_save_settings"))
        try:
            # --- Zakładka Główna ---
            config.settings["file_path"] = self.file_path.get()
            config.settings["split_rows"] = int(self.rows.get())
            config.settings["split_cols"] = int(self.cols.get())
            config.settings["overlap"] = float(self.overlap.get())
            config.settings["white_stripe"] = float(self.white_stripe.get())
            config.settings["add_white_to_overlap"] = self.add_white_to_overlap.get()
            config.settings["layout"] = self.layout.get()                  # zapisuj klucz układu
            config.settings["output_type"] = self.output_type.get()        # zapisuj klucz typu wyjścia
            config.settings["registration_mark_type"] = self.reg_mark.get()  # zapisuj klucz znacznika
            config.settings["barcode_type"] = self.barcode_type.get()      # zapisuj klucz typu kodu
            config.settings["bryt_order"] = self.bryt_order.get()          # zapisuj klucz kolejności brytów
            config.settings["enable_reg_marks"] = self.enable_reg_marks.get()
            config.settings["enable_barcode"] = self.enable_barcode.get()
            config.settings["enable_separation_lines"] = self.enable_separation_lines.get()
            config.settings["add_line_at_start"] = self.add_line_at_start.get()
            config.settings["add_line_at_end"] = self.add_line_at_end.get()
            config.settings["output_dir"] = self.output_dir.get()
            config.settings["slice_rotation"] = int(self.slice_rotation.get())
            config.settings["create_order_folder"] = self.create_order_folder.get() 
            config.settings["barcode_text_position"] = self.barcode_text_position.get() 
            config.settings["draw_slice_border"] = self.draw_slice_border.get() # <-- DODAJ TĘ LINIĘ

            # --- POCZĄTEK ZMIAN RAM/HDD---
            config.settings["processing_mode"] = self.processing_mode.get()
            # --- KONIEC ZMIAN ---


            # --- Zakładka Ustawienia ---
            if "graphic_settings" not in config.settings:
                config.settings["graphic_settings"] = {}
            for key, var in self.graphic_vars.items():
                try:
                    config.settings["graphic_settings"][key] = float(var.get())
                except ValueError:
                    config.settings["graphic_settings"][key] = 0.0
                    self.logger.warning(translate("logger_invalid_value").format(key=key))

            config.settings["logging_mode"] = self.logging_mode.get()  # zapisuj klucz trybu logowania
            config.settings["log_file_path"] = self.log_file_path.get()  # bez zmian
            config.settings["logging_level"] = self.logging_level.get()  # bez zmian

            if "code_settings" not in config.settings["graphic_settings"]:
                config.settings["graphic_settings"]["code_settings"] = {}
            code_settings_target = config.settings["graphic_settings"]["code_settings"]

            for code_type, output_types in self.code_gui_vars.items():
                if code_type not in code_settings_target:
                    code_settings_target[code_type] = {}
                for output_type_key, layouts in output_types.items():
                    if output_type_key not in code_settings_target[code_type]:
                        code_settings_target[code_type][output_type_key] = {}
                    for layout_key, params in layouts.items():
                        if layout_key not in code_settings_target[code_type][output_type_key]:
                            code_settings_target[code_type][output_type_key][layout_key] = {}
                        for param_key, var in params.items():
                            value_str = var.get()
                            target_dict = code_settings_target[code_type][output_type_key][layout_key]
                            if param_key == "anchor":
                                target_dict[param_key] = value_str
                            else:
                                try:
                                    target_dict[param_key] = float(value_str)
                                except ValueError:
                                    target_dict[param_key] = 0.0
                                    self.logger.warning(translate("logger_invalid_value").format(key=param_key))

            self.logger.debug(translate("logger_end_save_settings"))
            return True

        except ValueError as e:
            messagebox.showerror(translate("conversion_error_title"), translate("conversion_error").format(error=e))
            self.logger.error(translate("logger_conversion_error").format(error=e))
            return False
        except Exception as e:
            messagebox.showerror(translate("settings_save_error_title"), translate("settings_save_error").format(error=e))
            self.logger.error(translate("logger_unexpected_save_error").format(error=e), exc_info=True)
            return False

    def generate_pdf(self):
        """Uruchamia proces generowania PDF w osobnym wątku, aby nie blokować GUI."""
        self.logger.info(translate("logger_start_pdf"))
        if not self.save_settings_local():
            self.logger.warning(translate("logger_pdf_save_error"))
            return

        # Walidacja ścieżek przed uruchomieniem wątku
        output_directory = self.output_dir.get()
        input_file = self.file_path.get()

        if self.create_order_folder.get():
            if input_file and os.path.exists(input_file):
                order_folder_name = os.path.splitext(os.path.basename(input_file))[0]
                output_directory = os.path.join(output_directory, order_folder_name)
                try:
                    os.makedirs(output_directory, exist_ok=True)
                    self.logger.info(f"Utworzono lub potwierdzono istnienie folderu zamówienia: {output_directory}")
                except OSError as e:
                    messagebox.showerror(translate("error"), f"Nie można utworzyć folderu zamówienia: {output_directory}\nBłąd: {e}")
                    return
            else:
                messagebox.showwarning(translate("error"), "Aby utworzyć folder zamówienia, najpierw wybierz poprawny plik wejściowy.")
                return
        
        if not output_directory or not os.path.isdir(output_directory):
            messagebox.showerror(translate("error_output_dir_title"), translate("error_output_dir").format(directory=output_directory))
            self.logger.warning(translate("logger_invalid_output_dir").format(directory=output_directory))
            return

        if not input_file or not os.path.isfile(input_file):
            messagebox.showerror(translate("error_input_file_title"), translate("error_input_file").format(file=input_file))
            self.logger.warning(translate("logger_invalid_input_file").format(file=input_file))
            return
            
        # Natychmiastowe uruchomienie paska postępu
        home_ui = self.home_tab_instance
        home_ui.start_progress(home_ui.output_preview_canvas, home_ui.output_preview_progress)

        def worker():
            """Funkcja, która zostanie wykonana w osobnym wątku."""
            try:
                # Długotrwała operacja cięcia plików
                result = self.splitter.split(
                    input_pdf_path=config.settings.get("file_path"),
                    output_dir=output_directory,
                    overlap=config.settings.get("overlap", 20.0),
                    rows=config.settings.get("split_rows", 2),
                    cols=config.settings.get("split_cols", 5),
                    output_type=config.settings.get("output_type", "output_common_sheet"),
                    layout=config.settings.get("layout", "layout_vertical"),
                    enable_separation_lines=config.settings.get("enable_separation_lines", True),
                    add_line_at_start=config.settings.get("add_line_at_start", False),
                    add_line_at_end=config.settings.get("add_line_at_end", False),
                    enable_reg_marks=config.settings.get("enable_reg_marks", True),
                    registration_mark_type=config.settings.get("registration_mark_type", "reg_mark_cross"),
                    enable_barcode=config.settings.get("enable_barcode", True),
                    barcode_type=config.settings.get("barcode_type", "code_qr"),
                    bryt_order=config.settings.get("bryt_order", "bryt_order_1"),
                    white_stripe=config.settings.get("white_stripe", 10.0),
                    add_white_to_overlap=config.settings.get("add_white_to_overlap", True)
                )
                # Umieść wynik w kolejce, aby główny wątek mógł go odczytać
                self.work_queue.put(result)
            except Exception as e:
                # W razie błędu w wątku, również przekaż go do kolejki
                self.work_queue.put(e)

        # Uruchomienie wątku
        threading.Thread(target=worker, daemon=True).start()
        # Rozpoczęcie cyklicznego sprawdzania kolejki
        self.process_queue()

    def process_queue(self):
        """
        Sprawdza kolejkę z wynikiem pracy wątku.
        Obsługuje wyjątki oraz sukces (True). Po sukcesie pobiera ścieżki
        z instancji splittera, buduje słownik _last_result_paths i wywołuje podgląd.
        """
        try:
            item = self.work_queue.get_nowait()

            # --- JEST ELEMENT W KOLEJCE ---
            home_ui = getattr(self, "home_tab_instance", None)
            if home_ui:
                home_ui.stop_progress(home_ui.output_preview_canvas, home_ui.output_preview_progress)

            if isinstance(item, Exception):
                self.logger.error(f"Błąd wątku generowania PDF: {item}", exc_info=True)
                messagebox.showerror(translate("error"), str(item))
                # Nie kontynuujemy pętli po błędzie, czekamy na następną akcję użytkownika
                return

            if item is True:
                # Sukces - pobierz dane z instancji splittera i ustawień
                generated_files = self.splitter.generated_output_files
                mode = self.output_type.get()
                layout = self.layout.get()

                common_sheet_path = None
                separate_files_paths = []
                for f in generated_files:
                    if "_separated.pdf" in os.path.basename(f):
                        common_sheet_path = f
                    else:
                        separate_files_paths.append(f)
                
                separate_files_paths.sort()

                # Ustal preferowany plik do podglądu
                preferred = None
                if mode in ("output_common_sheet", "output_both") and common_sheet_path:
                    preferred = common_sheet_path
                elif separate_files_paths:
                    preferred = separate_files_paths[0]

                # Zbuduj słownik wyników
                self._last_result_paths = {
                    "mode": mode,
                    "layout": layout,
                    "common_sheet_path": common_sheet_path,
                    "separate_files": separate_files_paths,
                    "separate_first_path": separate_files_paths[0] if separate_files_paths else None,
                    "preview_preferred": preferred,
                }

                self.logger.info("Generowanie PDF zakończone sukcesem.")
                
                # Wywołaj aktualizację podglądu z nowymi danymi
                if home_ui:
                    home_ui.update_output_preview(self._last_result_paths, manage_progress=False)
            
            # Po przetworzeniu elementu, nie restartuj pętli. Czekaj na kolejną akcję.
            return

        except Empty:
            # Kolejka pusta, kontynuuj sprawdzanie
            self.root.after(100, self.process_queue)

def load_profiles_from_file():
    """Wczytuje profile użytkownika z pliku JSON."""
    base_dir = get_base_dir()
    full_path = os.path.join(base_dir, PROFILES_FILE)
    logger.info(f"{translate('loading_profiles_from')}: {full_path}")
    try:
        # Próba otwarcia i odczytania pliku profili
        with open(full_path, "r", encoding="utf-8") as f:
            profiles = json.load(f)
        # Sprawdzenie, czy wczytane dane są słownikiem
        if not isinstance(profiles, dict):
            logger.warning(f"{translate('profiles_file')} {PROFILES_FILE} {translate('does_not_contain_dict')}.")
            return {} # Zwróć pusty słownik, jeśli format jest nieprawidłowy
        return profiles # Zwróć wczytane profile
    except FileNotFoundError:
        # Informacja, jeśli plik profili nie istnieje (normalne przy pierwszym uruchomieniu)
        logger.info(f"{translate('profiles_file')} {PROFILES_FILE} {translate('not_found_creating_new')}.")
        return {} # Zwróć pusty słownik, zostanie utworzony przy zapisie
    except json.JSONDecodeError as e:
        # Obsługa błędu, jeśli plik JSON jest uszkodzony
        messagebox.showerror(translate("error"), f"{translate('json_profiles_read_error')}:\n{e}\n\n{translate('file_will_be_overwritten')}")
        logger.error(f"{translate('json_decode_error_in_profiles')}: {e}")
        return {} # Zwróć pusty słownik, aby zapobiec dalszym błędom
    except Exception as e:
        # Obsługa innych nieoczekiwanych błędów odczytu
        messagebox.showerror(translate("error"), f"{translate('cannot_load_profiles_file')}:\n{e}")
        logger.error(f"{translate('unexpected_profiles_read_error')}: {e}", exc_info=True)
        return {} # Zwróć pusty słownik w razie błędu

        self.status_label.pack(side="left", padx=5, pady=5)

            self.status_label.config(text=translate("license_active"), foreground="green")

    def update_preview(self, manage_progress=True):
        """Aktualizuje podgląd pliku wejściowego, skalując go do rozmiaru płótna."""
        if self.preview_canvas is None:
            return

        if manage_progress:
            self.start_progress(self.preview_canvas, self.preview_progress)
        
        # Opóźnienie, aby dać czas na wyrenderowanie canvasu i uzyskanie jego prawidłowych wymiarów
        self.preview_canvas.after(100, self._update_preview_task, manage_progress)

    def _update_preview_task(self, manage_progress):
        """Zadanie właściwej aktualizacji podglądu, uruchamiane po ustabilizowaniu UI."""
        if self.preview_canvas is None:
            return
        canvas_w = self.preview_canvas.winfo_width()
        canvas_h = self.preview_canvas.winfo_height()

        if canvas_w < 2 or canvas_h < 2:
            if manage_progress:
                self.stop_progress(self.preview_canvas, self.preview_progress)
            self._show_message_on_canvas(self.preview_canvas, translate("msg_no_input_file"))
            return

        def task():
            try:
                path = self.app.file_path.get().strip()
                if not path or not os.path.exists(path):
                    if self.preview_canvas is not None:
                        self.preview_canvas.after(0, lambda: self._show_message_on_canvas(self.preview_canvas, translate("msg_no_input_file")))
                    return

                poppler_dir = get_poppler_path()
                # Generuj miniaturę o wymiarach pasujących do canvasu
                pages = convert_from_path(path, dpi=72, first_page=1, last_page=1, fmt='png', thread_count=4, size=(canvas_w, None), poppler_path=poppler_dir)
                if not pages:
                     raise ValueError("convert_from_path nie zwrócił żadnych stron.")
                img = pages[0]
                
                # Dodatkowe przeskalowanie, aby na pewno zmieścić się w pionie, zachowując proporcje
                img.thumbnail((canvas_w, canvas_h), Image.Resampling.LANCZOS)

                self.preview_image = ImageTk.PhotoImage(img)
                if self.preview_canvas is not None:
                    self.preview_canvas.after(0, lambda: self._display_image_and_grid(img))

            except Exception as e:
                self.logger.error(f"Błąd podczas aktualizacji podglądu wejściowego: {e}", exc_info=True)
                if self.preview_canvas is not None:
                    self.preview_canvas.after(0, lambda: self._show_message_on_canvas(self.preview_canvas, translate("msg_thumbnail_error")))
            finally:
                if manage_progress and self.preview_canvas is not None:
                    self.preview_canvas.after(0, lambda: self.stop_progress(self.preview_canvas, self.preview_progress))

        threading.Thread(target=task, daemon=True).start()

    def _display_image_and_grid(self, img):
        """Wyświetla obraz i siatkę na płótnie podglądu wejściowego."""
        if self.preview_canvas is None:
            return
        self.preview_canvas.delete("all")
        canvas_w = self.preview_canvas.winfo_width()
        canvas_h = self.preview_canvas.winfo_height()
        self.preview_canvas.create_image(canvas_w / 2, canvas_h / 2, image=self.preview_image)

        try:
            rows, cols = int(self.app.rows.get()), int(self.app.cols.get())
        except ValueError:
            rows, cols = 1, 1

        img_w, img_h = img.width, img.height
        x_offset = (canvas_w - img_w) / 2
        y_offset = (canvas_h - img_h) / 2

        cell_w, cell_h = img_w / cols, img_h / rows
        for i in range(1, cols):
            x = x_offset + i * cell_w
            self.preview_canvas.create_line(x, y_offset, x, y_offset + img_h, fill="red", dash=(4, 2))
        for j in range(1, rows):
            y = y_offset + j * cell_h
            self.preview_canvas.create_line(x_offset, y, x_offset + img_w, y, fill="red", dash=(4, 2))

        for r in range(rows):
            for c in range(cols):
                lbl = chr(65 + r) + str(c + 1)
                x = x_offset + c * cell_w + cell_w / 2
                y = y_offset + r * cell_h + cell_h / 2
                self._draw_text_with_outline(self.preview_canvas, x, y, lbl, ("Helvetica", 12, "bold"), "white", "black", offset=1)

    def _show_message_on_canvas(self, canvas, message):
        """Wyświetla komunikat na środku danego płótna."""
        canvas.delete("all")
        canvas_w = canvas.winfo_width()
        canvas_h = canvas.winfo_height()
        canvas.create_text(
            canvas_w / 2, canvas_h / 2,
            text=message, fill="gray", font=("Helvetica", 14), justify="center"
        )







    def _update_widget_labels(self):
        """Aktualizuje tekst na wszystkich zapisanych widgetach, które tego wymagają."""
        for widget, key in self.translatable_widgets:
            try:
                if widget.winfo_exists():
                    # Specjalna obsługa dla LabelFrame
                    if isinstance(widget, ttk.LabelFrame):
                        widget.configure(text=" " + translate(key) + " ")
                    else:
                        widget.configure(text=translate(key))
                    _apply_direction(widget)
            except Exception as e:
                logger.warning(f"Nie udało się zaktualizować widgetu dla klucza '{key}': {e}")

    def _select_output_preview_file(self, result_paths: dict) -> str | None:
        """
        Strategia wyboru pliku do podglądu WYJŚCIA na podstawie słownika z wynikami.
        """
        if not result_paths:
            return None

        preferred = result_paths.get("preview_preferred")
        if preferred and os.path.isfile(preferred):
            self.logger.debug(f"Output preview: using preferred path: {preferred}")
            return preferred

        mode = result_paths.get("mode")
        common_path = result_paths.get("common_sheet_path")
        if mode in ("output_common_sheet", "output_both") and common_path and os.path.isfile(common_path):
            self.logger.debug(f"Output preview: using common_sheet_path: {common_path}")
            return common_path

        separate_files = result_paths.get("separate_files", [])
        if mode == "output_separate_files" and separate_files:
            first_file = separate_files[0]
            if os.path.isfile(first_file):
                self.logger.debug(f"Output preview: using separate_files[0]: {first_file}")
                return first_file
        
        self.logger.warning("Could not select a suitable output file for preview.")
        return None

    def update_output_preview(self, result_paths: dict | None = None, manage_progress: bool = True):
        """
        Główna metoda uruchamiająca aktualizację podglądu wyjścia.
        Pobiera słownik z wynikami i zleca renderowanie miniatury.
        """
        if self.output_preview_canvas is None:
            self.logger.debug("Output preview canvas is not available, skipping update.")
            return

        if manage_progress:
            self.start_progress(self.output_preview_canvas, self.output_preview_progress)

        def _task():
            try:
                # Użyj przekazanych ścieżek lub pobierz ostatnie z instancji aplikacji
                paths = result_paths or getattr(self.app, "_last_result_paths", None)
                if not paths:
                    self.logger.debug("update_output_preview: No result paths available, clearing preview.")
                    self._show_output_preview_message(translate("msg_no_pdf_output"))
                    return

                # Wybierz jeden plik do wyświetlenia
                file_to_preview = self._select_output_preview_file(paths)

                if file_to_preview:
                    self._generate_preview_from_output([file_to_preview])
                else:
                    self.logger.warning("No output file selected for preview.")
                    self._show_output_preview_message(translate("msg_no_pdf_output"))
            finally:
                if manage_progress:
                    self.stop_progress(self.output_preview_canvas, self.output_preview_progress)

        self.output_preview_canvas.after(120, _task)




    def _pick_files_for_output_preview(self) -> list[str]:
        """
        Zwraca listę ścieżek plików do podglądu WYJŚCIA.
        Zasady:
        - output_common_sheet: 1 plik – wspólny arkusz
        - output_separate_files: do 9 pierwszych pociętych brytów (sortowane alfanumerycznie)
        - output_both: ZAWSZE podgląd po wspólnym arkuszu
        """
        import os, glob

        output_dir = (self.app.output_dir.get() or "").strip()
        if not output_dir:
            self.logger.debug("Output dir pusty – brak plików do podglądu.")
            return []

        # normalizacja ścieżki (eliminuje mieszanie slash/backslash)
        output_dir = os.path.normpath(output_dir)

        output_type = getattr(self, "output_type_var", None)
        output_type = output_type.get() if output_type else getattr(self, "output_type", "output_common_sheet")

        layout_var = getattr(self, "layout_var", None)
        layout = layout_var.get() if layout_var else getattr(self, "layout", "layout_vertical")
        lay_char = "v" if layout == "layout_vertical" else "h"

        base_name = (self.app.order_base_name_var.get() or "").strip()  # np. "billboard3"
        if not base_name:
            # jeśli nie mamy nazwy zamówienia, spróbuj wykryć jedyny wspólny arkusz w katalogu
            pattern_auto = os.path.join(output_dir, f"*_{lay_char}_separated.pdf")
            candidates = sorted(glob.glob(pattern_auto))
            if output_type in ("output_common_sheet", "output_both"):
                if candidates:
                    self.logger.debug(f"Auto-detekcja wspólnego arkusza: {candidates[0]}")
                    return [candidates[0]]
            # dla osobnych spróbujmy zebrać *.pdf
            pattern_sep = os.path.join(output_dir, f"*_*_{lay_char}.pdf")
            files = sorted(glob.glob(pattern_sep))[:9]
            return files

        # Nazwa pliku wspólnego arkusza wg konwencji z logów:
        #   {base_name}_{v|h}_separated.pdf  (np. billboard3_v_separated.pdf)
        common_sheet_path = os.path.normpath(os.path.join(output_dir, f"{base_name}_{lay_char}_separated.pdf"))

        if output_type == "output_common_sheet":
            if os.path.isfile(common_sheet_path):
                return [common_sheet_path]
            else:
                self.logger.debug(f"Brak wspólnego arkusza pod: {common_sheet_path}")
                return []

        if output_type == "output_both":
            # zawsze pokazuj wspólny arkusz jako miniaturę
            if os.path.isfile(common_sheet_path):
                return [common_sheet_path]
            else:
                self.logger.debug(f"Brak wspólnego arkusza (output_both) pod: {common_sheet_path}")
                return []

        # output_separate_files
        # Przykładowe nazwy z logów: 0A01_0A01_h.pdf, 0B05_0B05_v.pdf, ...
        pattern = os.path.join(output_dir, f"*_*_{lay_char}.pdf")
        files = sorted(glob.glob(pattern))[:9]
        if not files:
            self.logger.debug(f"Nie znaleziono osobnych plików wg wzorca: {pattern}")
        return files




    def _open_pdf_first_page_png(self, pdf_path: str, width_px: int, height_px: int):
        """
        Renderuje 1. stronę PDF do obrazu (Pillow) o mniej więcej zadanym rozmiarze.
        """
        import os, math
        from pdf2image import convert_from_path

        pdf_path = os.path.normpath(pdf_path)

        if not os.path.isfile(pdf_path):
            self.logger.debug(f"PDF nie istnieje: {pdf_path}")
            return None

        # dobranie DPI tak, aby szerokość ~ width_px (z zachowaniem proporcji)
        # najpierw szybki render niskim dpi, żeby poznać proporcje
        poppler_dir = get_poppler_path()
        pages_lo = convert_from_path(pdf_path, dpi=36, first_page=1, last_page=1, fmt='png',
                                    poppler_path=poppler_dir)
        if not pages_lo:
            return None

        w0, h0 = pages_lo[0].width, pages_lo[0].height
        if w0 <= 0 or h0 <= 0:
            return None

        scale = width_px / float(w0)
        dpi = max(36, min(220, int(36 * scale)))  # bezpieczny zakres dpi

        pages = convert_from_path(pdf_path, dpi=dpi, first_page=1, last_page=1, fmt='png',
                                poppler_path=poppler_dir)
        if not pages:
            return None

        img = pages[0]
        img.thumbnail((width_px, height_px))
        return img




    def _render_output_preview_bitmap(self, files: list[str]):
        """
        Buduje obrazek do podglądu:
        - 1 plik → render 1. strony PDF
        - >=2 pliki → mozaika 3×3 z pierwszych 9 PDF-ów (1. strony)
        """
        from PIL import Image

        if not files:
            return None

        # docelowy rozmiar „roboczy” – i tak potem thumbnail do canvasa
        max_w = max_h = 900

        try:
            if len(files) == 1:
                return self._open_pdf_first_page_png(files[0], max_w, max_h)
            else:
                cols = rows = 3
                cell = 280
                spacing = 16
                W = cols * cell + (cols - 1) * spacing
                H = rows * cell + (rows - 1) * spacing
                canvas = Image.new("RGB", (W, H), "white")

                i = 0
                for r in range(rows):
                    for c in range(cols):
                        if i >= len(files):
                            break
                        try:
                            im = self._open_pdf_first_page_png(files[i], cell, cell)
                            if im is not None:
                                # środek komórki
                                x = c * (cell + spacing)
                                y = r * (cell + spacing)
                                im.thumbnail((cell, cell))
                                # centrowanie wewnątrz komórki
                                ox = x + (cell - im.width) // 2
                                oy = y + (cell - im.height) // 2
                                canvas.paste(im, (ox, oy))
                        except Exception as e:
                            self.logger.debug(f"Miniatura pominięta ({files[i]}): {e}")
                        i += 1
                return canvas
        except Exception as e:
            self.logger.error(f"Render bitmapy podglądu nie powiódł się: {e}", exc_info=True)
            return None



    def _update_output_preview_task(self, manage_progress: bool):
        """Właściwa logika renderu miniatury w wątku roboczym + bezpieczny powrót do GUI."""
        if self.output_preview_canvas is None:
            return

        canvas_w = max(20, int(self.output_preview_canvas.winfo_width() or 0))
        canvas_h = max(20, int(self.output_preview_canvas.winfo_height() or 0))

        # Wszystko w wątku roboczym (render PDF -> obraz)
        def worker():
            from PIL import Image, ImageTk
            try:
                layout = self.app.layout.get()
                output_type = self.app.output_type.get()
                output_dir = self.app.output_dir.get()

                # Prepare a dict as expected by _select_output_preview_file
                result_paths = {
                    "mode": output_type,
                    "common_sheet_path": None,  # You may fill this if you have the path
                    "separate_files": None,     # You may fill this if you have the list
                    "preview_preferred": None
                }
                pdf_path = self._select_output_preview_file(result_paths)
                if not pdf_path or not os.path.exists(pdf_path):
                    # nie czyścimy wejścia! tylko komunikat na wyjściowym
                    if self.output_preview_canvas is not None:
                        self.output_preview_canvas.after(0, lambda: self._show_output_preview_message(translate("msg_no_pdf_output")))
                    return

                # render 1. strony do pixmapy (pymupdf)
                try:
                    doc = fitz.open(str(pdf_path))
                    if not doc.page_count:
                        raise RuntimeError(f"PDF has no pages: {pdf_path}")
                    page = doc.load_page(0)
                    pix = page.get_pixmap(dpi=96)
                    img = Image.frombytes("RGB", (pix.width, pix.height), pix.samples)
                    doc.close()
                except Exception as e:
                    self.logger.error(f"Output preview render failed for '{pdf_path}': {e}", exc_info=True)
                    if self.output_preview_canvas is not None:
                        self.output_preview_canvas.after(0, lambda: self._show_output_preview_message(translate("msg_thumbnail_error")))
                    return

                # skalowanie do rozmiaru canvasa
                img.thumbnail((canvas_w - 10, canvas_h - 10), Image.Resampling.LANCZOS)
                self.output_preview_image = ImageTk.PhotoImage(img)

                # malowanie w GUI
                if self.output_preview_canvas is not None:
                    self.output_preview_canvas.after(0, self._update_canvas_image)

            finally:
                if manage_progress and self.output_preview_canvas is not None:
                    self.output_preview_canvas.after(0, lambda: self.stop_progress(self.output_preview_canvas, self.output_preview_progress))

        threading.Thread(target=worker, daemon=True).start()

    
    
    def _create_composite_preview_grid(self, files: list[str]) -> Image.Image:
        """
        (Opcjonalny helper) Siatka miniaturek dla wielu PDF-ów.
        Nie jest używany dla 'output_both' (tam zawsze wspólny arkusz), ale zostawiamy na przyszłość.
        """
        if not files:
            return Image.new("RGB", (1, 1), "white")

        thumbs = []
        max_thumb_dim = 110
        spacing = 6

        # Wczytanie miniatur
        for p in files[:9]:  # max 3x3
            try:
                doc = fitz.open(p)
                if doc.page_count:
                    pix = doc.load_page(0).get_pixmap(dpi=40)
                    img = Image.frombytes("RGB", (pix.width, pix.height), pix.samples)
                    img.thumbnail((max_thumb_dim, max_thumb_dim), Image.Resampling.LANCZOS)
                    thumbs.append(img)
                doc.close()
            except Exception as e:
                self.logger.debug(f"Thumbnail skip ({p}): {e}")

        if not thumbs:
            return Image.new("RGB", (1, 1), "white")

        # Układ 3x3
        cols = rows = 3
        cell = max_thumb_dim
        W = cols * cell + (cols - 1) * spacing
        H = rows * cell + (rows - 1) * spacing
        canvas = Image.new("RGB", (W, H), "white")

        i = 0
        for r in range(rows):
            for c in range(cols):
                if i >= len(thumbs):
                    break
                x = c * (cell + spacing)
                y = r * (cell + spacing)
                # wycentruj w komórce
                tw, th = thumbs[i].size
                ox = x + (cell - tw) // 2
                oy = y + (cell - th) // 2
                canvas.paste(thumbs[i], (ox, oy))
                i += 1

        return canvas
    
    
    
    def _generate_preview_from_output(self, files: list[str], manage_progress: bool = False):
        """
        Renderuje miniaturę pierwszej strony podanego pliku (files[0]) na canvasie wyjścia.
        """
        if not files or not files[0]:
            self.logger.warning("generate_preview_from_output: No file path provided.")
            self._show_output_preview_message(translate("msg_no_pdf_output"))
            return

        pdf_path = files[0]
        canvas = self.output_preview_canvas
        if not canvas:
            return

        canvas_w = canvas.winfo_width()
        canvas_h = canvas.winfo_height()

        if canvas_w <= 2 or canvas_h <= 2:
            canvas.after(120, lambda: self._generate_preview_from_output(files, manage_progress))
            return

        def _render_task():
            try:
                self.logger.debug(f"Rendering output preview for: {pdf_path}")
                doc = fitz.open(pdf_path)
                if not doc.page_count:
                    raise ValueError(f"PDF file has no pages: {pdf_path}")
                
                page = doc.load_page(0)
                pix = page.get_pixmap(dpi=96)
                img = Image.frombytes("RGB", [pix.width, pix.height], pix.samples)
                doc.close()

                img.thumbnail((canvas_w - 10, canvas_h - 10), Image.Resampling.LANCZOS)
                self.output_preview_image = ImageTk.PhotoImage(img)

                def _update_gui():
                    if self.output_preview_canvas:
                        self.output_preview_canvas.delete("all")
                        self.output_preview_canvas.create_image(
                            canvas_w / 2, canvas_h / 2, image=self.output_preview_image, anchor="center"
                        )
                
                canvas.after(0, _update_gui)

            except Exception as e:
                self.logger.error(f"Failed to render output preview for '{pdf_path}': {e}", exc_info=True)
                self._show_output_preview_message(translate("msg_thumbnail_error"))

        # Uruchom renderowanie w osobnym wątku, aby nie blokować GUI
        threading.Thread(target=_render_task, daemon=True).start()

    def _generate_and_display_preview(self, manage_progress):
        """Metoda pomocnicza do generowania szybkiego podglądu w tle."""
        def task():
            try:
                path = self.app.file_path.get().strip()
                if not path or not os.path.exists(path):
                    if self.output_preview_canvas is not None:
                        self.output_preview_canvas.after(0, lambda: self._show_message_on_canvas(self.output_preview_canvas, translate("msg_no_input_file")))
                    return

                current_settings = {
                    "split_rows": self.app.rows.get(), "split_cols": self.app.cols.get(),
                    "layout": self.app.layout.get(), "bryt_order": self.app.bryt_order.get(),
                    "slice_rotation": self.app.slice_rotation.get(), "overlap": self.app.overlap.get(),
                    "white_stripe": self.app.white_stripe.get(), "add_white_to_overlap": self.app.add_white_to_overlap.get()
                }

                preview_image = self.app.splitter.generate_fast_preview(path, current_settings)

                if preview_image:
                    if self.output_preview_canvas is not None:
                        canvas_w = self.output_preview_canvas.winfo_width()
                        canvas_h = self.output_preview_canvas.winfo_height()
                        if canvas_w < 2 or canvas_h < 2:
                            canvas_w = canvas_h = OUTPUT_PREVIEW_CANVAS_SIZE
                    else:
                        canvas_w = canvas_h = OUTPUT_PREVIEW_CANVAS_SIZE
                    
                    preview_image.thumbnail((canvas_w, canvas_h), Image.Resampling.LANCZOS)
                    self.output_preview_image = ImageTk.PhotoImage(preview_image)
                    
                    if self.output_preview_canvas is not None:
                        self.output_preview_canvas.after(0, self._update_canvas_image)
                else:
                    if self.output_preview_canvas is not None:
                        self.output_preview_canvas.after(0, self._show_preview_error)
            except Exception as e:
                self.logger.error(f"Błąd podczas generowania podglądu wyjściowego: {e}", exc_info=True)
                if self.output_preview_canvas is not None:
                    self.output_preview_canvas.after(0, self._show_preview_error)
            finally:
                if manage_progress and self.output_preview_canvas is not None:
                    self.output_preview_canvas.after(0, lambda: self.stop_progress(self.output_preview_canvas, self.output_preview_progress))

        threading.Thread(target=task, daemon=True).start()






    def _create_composite_preview(self, files: list) -> Image.Image:
        """Tworzy złożony obraz z miniatur wielu plików PDF."""
        thumbs = []
        max_thumb_dim = 100  # maksymalny rozmiar pojedynczej miniatury
        spacing = 5

        if not files:
            return Image.new("RGB", (1, 1), "white")

        # Generowanie miniatur wątkowo dla przyspieszenia (równoległe wczytywanie pierwszych stron)
        try:
            from concurrent.futures import ThreadPoolExecutor
            max_workers = min(4, len(files))
            def load_thumb(file_path):
                try:
                    doc = fitz.open(file_path)
                    if doc.page_count > 0:
                        page = doc.load_page(0)
                        pix = page.get_pixmap(dpi=30)
                        img = Image.frombytes("RGB", (pix.width, pix.height), pix.samples)
                        doc.close()
                        img.thumbnail((max_thumb_dim, max_thumb_dim), Image.Resampling.LANCZOS)
                        return img
                except Exception as e:
                    self.app.logger.error(f"{translate('cannot_load_thumbnail')} {file_path}: {e}")
                return None

            with ThreadPoolExecutor(max_workers=max_workers) as executor:
                futures = [executor.submit(load_thumb, path) for path in files]
                # Pobierz wyniki zachowując kolejność oryginalnej listy plików
                for future in futures:
                    img = future.result()
                    if img:
                        thumbs.append(img)
        except Exception as e:
            # W razie błędów wielowątkowości – powrót do przetwarzania sekwencyjnego
            self.app.logger.debug(f"Threaded thumbnail generation failed, falling back to sequential: {e}")
            thumbs = []
            for file_path in files:
                try:
                    doc = fitz.open(file_path)
                    if doc.page_count > 0:
                        page = doc.load_page(0)
                        pix = page.get_pixmap(dpi=30)
                        img = Image.frombytes("RGB", (pix.width, pix.height), pix.samples)
                        doc.close()
                        img.thumbnail((max_thumb_dim, max_thumb_dim), Image.Resampling.LANCZOS)
                        thumbs.append(img)
                except Exception as e2:
                    self.app.logger.error(f"{translate('cannot_load_thumbnail')} {file_path}: {e2}")

        if not thumbs:
            return Image.new("RGB", (1, 1), "white")

        # Ułożenie miniatur w jednym wierszu na wspólnym obrazie podglądu
        total_width = sum(img.width for img in thumbs) + spacing * (len(thumbs) - 1)
        max_height = max(img.height for img in thumbs)
        composite_img = Image.new("RGB", (total_width, max_height), "lightgrey")

        x_offset = 0
        for img in thumbs:
            y_offset = (max_height - img.height) // 2
            composite_img.paste(img, (x_offset, y_offset))
            x_offset += img.width + spacing

        return composite_img

                for out_key, layouts in output_types.items():
                    code_settings_target[code_type].setdefault(out_key, {})
                    for layout_key, params in layouts.items():
                        code_settings_target[code_type][out_key].setdefault(layout_key, {})
                        for param_key, var in params.items():
                            value = var.get()
                            target = code_settings_target[code_type][out_key][layout_key]
                            if param_key == "anchor":
                                target[param_key] = value
                            else:
                                try:
                                    target[param_key] = float(value)
                                except ValueError:
                                    target[param_key] = 0.0
                                    self.logger.warning(translate("logger_invalid_value").format(key=param_key))

            self.logger.debug(translate("logger_end_save_settings"))
            return True

        except ValueError as e:
            messagebox.showerror(
                translate("conversion_error_title"),
                translate("conversion_error").format(error=e)
            )
            self.logger.error(translate("logger_conversion_error").format(error=e))
            return False

        except Exception as e:
            messagebox.showerror(
                translate("settings_save_error_title"),
                translate("settings_save_error").format(error=e)
            )
            self.logger.error(translate("logger_unexpected_save_error").format(error=e), exc_info=True)
            return False

    def _update_widget_labels(self):
        """Aktualizuje tekst na wszystkich zapisanych widgetach, które tego wymagają."""
        for widget, key in self.translatable_widgets:
            try:
                if widget.winfo_exists():
                    if isinstance(widget, ttk.LabelFrame):
                        widget.configure(text=" " + translate(key) + " ")
                    else:
                        widget.configure(text=translate(key))
                    _apply_direction(widget)
            except Exception as e:
                logger.warning(f"Nie udało się zaktualizować widgetu dla klucza '{key}': {e}")

    def update_combobox_values(self):
        """Aktualizuje wartości w komboboxach i etykiety ramek po zmianie języka."""
        self._update_widget_labels()

        if 'logging_mode' in self.comboboxes:
            current_key = config.settings.get("logging_mode", "logging_console")
            new_values = [translate("logging_console"), translate("logging_file"), translate("logging_both")]
            self.comboboxes['logging_mode']['values'] = new_values
            self.comboboxes['logging_mode'].set(translate(current_key))
            _apply_direction(self.comboboxes['logging_mode'])

        if 'barcode_text_position' in self.comboboxes:
            current_key = self.app.barcode_text_position.get()
            new_values = [translate("barcode_text_bottom"), translate("barcode_text_side"), translate("barcode_text_none")]
            self.comboboxes['barcode_text_position']['values'] = new_values
            key_map = {
                "bottom": translate("barcode_text_bottom"),
                "side": translate("barcode_text_side"),
                "none": translate("barcode_text_none")
            }
            self.comboboxes['barcode_text_position'].set(key_map.get(current_key, translate("barcode_text_side")))
            _apply_direction(self.comboboxes['barcode_text_position'])
        
        self.logger.debug("Odświeżono wartości comboboxów i etykiety w zakładce Ustawienia.")

def get_base_dir():
    """Określa bazowy katalog aplikacji."""
    if getattr(sys, 'frozen', False):
        # Uruchomiono jako .exe (zamrożone przez PyInstaller)
        return sys._MEIPASS  # Używamy sys._MEIPASS, gdzie PyInstaller rozpakowuje zasoby
    else:
        # Uruchomiono jako skrypt .py
        return os.path.dirname(os.path.abspath(__file__))

def _on_mousewheel(widget: tk.Canvas):
    """
    Ustawia obsługę kółka myszy dla przewijania wskazanego widgetu Canvas.
    Dla Windows/macOS oraz Linux – rejestruje bindingi dla odpowiednich zdarzeń.
    """
    def _mousewheel(event):
        # Dla Windows i macOS (event.delta)
        if event.delta:
            widget.yview_scroll(int(-1 * (event.delta / 120)), "units")
        # Dla Linux (zdarzenia Button-4/5)
        elif event.num == 4:  # Scroll up
            widget.yview_scroll(-1, "units")
        elif event.num == 5:  # Scroll down
            widget.yview_scroll(1, "units")

    # Bind dla różnych platform
    widget.bind("<MouseWheel>", _mousewheel)
    widget.bind("<Button-4>", _mousewheel)
    widget.bind("<Button-5>", _mousewheel)

def _apply_direction(widget, *, for_multiline=False):
    try:
        from lang_manager import text_align, text_justify, is_rtl
        align = text_align()
        justify = text_justify()
        # ttk.Label/ttk.Button: anchor steruje położeniem w komórce; justify – wyrównanie tekstu wieloliniowego
        if hasattr(widget, "configure"):
            # anchor dla etykiet/ramek/przycisków
            if "anchor" in widget.configure():
                widget.configure(anchor="e" if is_rtl() else "w")
            # justify dla tekstu wieloliniowego
            if for_multiline and "justify" in widget.configure():
                widget.configure(justify=justify)
        # Entry/Text – ustaw kierunek logiczny poprzez justify/right-left
        if widget.winfo_class() in ("TEntry", "Entry"):
            if "justify" in widget.configure():
                widget.configure(justify="right" if is_rtl() else "left")
        # Combobox: wyrównanie tekstu
        if widget.winfo_class() in ("TCombobox",):
            if "justify" in widget.configure():
                widget.configure(justify="right" if is_rtl() else "left")
    except Exception as e:
        logger = logging.getLogger('BillboardSplitter')
        logger.debug(f"_apply_direction failed: {e}")

    def __init__(self, config_module):
        self.config = config_module
        self.logger = self.setup_logging()
        self.trial_mode = verify_license(logger=self.logger)
        if self.trial_mode:
            self.logger.info(LANG["log_trial_mode_active"])
        else:
            self.logger.info(LANG["log_trial_mode_inactive"])

        self.generated_output_files = []
        self.mm = 2.8346456693  # 1 mm in points
        self.update_graphic_settings()

    def update_graphic_settings(self):
        """Przeładowuje ustawienia graficzne z modułu config."""
        try:
            gs = self.config.settings.get("graphic_settings", {})
            self.margin_top = gs.get("margin_top", 2.0) * self.mm
            self.margin_bottom = gs.get("margin_bottom", 2.0) * self.mm
            self.margin_left = gs.get("margin_left", 2.0) * self.mm
            self.margin_right = gs.get("margin_right", 2.0) * self.mm
            self.mark_width = gs.get("reg_mark_width", 5.0) * self.mm
            self.mark_height = gs.get("reg_mark_height", 5.0) * self.mm

            self.reg_mark_line_width_white = gs.get("reg_mark_white_line_width", 0.5) * self.mm
            self.reg_mark_line_width_black = gs.get("reg_mark_black_line_width", 0.7) * self.mm
            self.border_line_width = 0.1 * self.mm

            self.logger.debug(f"Marginesy ustawione na (G/D/L/P): {self.margin_top/self.mm}/{self.margin_bottom/self.mm}/{self.margin_left/self.mm}/{self.margin_right/self.mm} mm")
        except Exception as e:
            self.logger.error(LANG["log_graphic_settings_error"].format(e), exc_info=True)

    def setup_logging(self):
        # Ta funkcja pozostaje bez zmian, jest już poprawna.
        logger = logging.getLogger('BillboardSplitter')
        if logger.hasHandlers():
            logger.handlers.clear()
        logger.propagate = False
        try:
            log_level_str = self.config.settings.get("logging_level", "INFO").upper()
            log_level = getattr(logging, log_level_str, logging.INFO)
            logger.setLevel(log_level)
            formatter = logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s')
            logging_mode = self.config.settings.get("logging_mode", "console")
            log_file_path = self.config.settings.get("log_file_path", "logs/app.log")
            if logging_mode in ["logging_console", "logging_both"]:
                ch = logging.StreamHandler(sys.stdout)
                ch.setFormatter(formatter)
                logger.addHandler(ch)
            if logging_mode in ["logging_file", "logging_both"]:
                log_dir = os.path.dirname(log_file_path)
                os.makedirs(log_dir, exist_ok=True)
                fh = TimedRotatingFileHandler(log_file_path, when="midnight", interval=1, backupCount=7, encoding="utf-8")
                fh.setFormatter(formatter)
                logger.addHandler(fh)
        except Exception as e:
            logging.basicConfig(level=logging.WARNING)
            logger = logging.getLogger('BillboardSplitter')
            logger.error(f"Krytyczny błąd podczas konfiguracji logowania: {e}", exc_info=True)
        return logger

    def _load_input_as_pdf(self, input_path: str) -> tuple[fitz.Document | None, bool]:
        """Wczytuje plik wejściowy (PDF, JPG, TIFF, PNG) i konwertuje go na obiekt PDF PyMuPDF."""
        self.logger.debug(f"Ładowanie pliku: {input_path}")
        ext = os.path.splitext(input_path)[1].lower()

        if ext == ".pdf":
            try:
                doc = fitz.open(input_path)
                if not doc.page_count:
                    self.logger.error("Plik PDF nie zawiera żadnych stron.")
                    doc.close()
                    return None, False
                return doc, False
            except Exception as e:
                self.logger.error(f"Nie można otworzyć pliku PDF: {e}", exc_info=True)
                return None, False

        if ext in [".jpg", ".jpeg", ".tif", ".tiff", ".png"]:
            try:
                with Image.open(input_path) as img:
                    if img.mode == 'CMYK':
                        img = img.convert('RGB')
                    dpi = img.info.get('dpi', (300, 300))
                    dpi_val = dpi[0] if isinstance(dpi, tuple) else dpi
                    dpi_val = float(dpi_val) if dpi_val > 0 else 300.0
                    factor = 72.0 / dpi_val
                    width_pts, height_pts = img.width * factor, img.height * factor
                    pdf_doc = fitz.open()
                    page = pdf_doc.new_page(width=width_pts, height=height_pts)
                    img_bytes = io.BytesIO()
                    img.save(img_bytes, format='PNG')
                    img_bytes.seek(0)
                    page.insert_image(page.rect, stream=img_bytes)
                    self.logger.debug(f"Pomyślnie przekonwertowano obraz na PDF w pamięci.")
                    return pdf_doc, True
            except Exception as e:
                self.logger.error(f"Błąd konwersji obrazu na PDF: {e}", exc_info=True)
                return None, False

        self.logger.error(f"Nieobsługiwany format pliku: {ext}")
        return None, False

    def calculate_dimensions(self, overlap, rows, cols, white_stripe, add_white_to_overlap=False):
        """Oblicza wymiary brytów i rdzeni. Działa na punktach (pts)."""
        W_mm, H_mm = self.width / self.mm, self.height / self.mm
        self.use_white_stripe = add_white_to_overlap
        self.white_stripe = abs(white_stripe) if self.use_white_stripe else 0.0
        self.overlap_value = abs(overlap)
        self.effective_overlap = self.overlap_value + self.white_stripe
        S_x_mm = (W_mm + (cols - 1) * self.effective_overlap) / cols if cols > 0 else 0
        S_y_mm = (H_mm + (rows - 1) * self.effective_overlap) / rows if rows > 0 else 0
        self.slice_width = S_x_mm * self.mm
        self.slice_height = S_y_mm * self.mm
        self.core_slice_width = self.slice_width - self.effective_overlap * self.mm
        self.core_slice_height = self.slice_height - self.effective_overlap * self.mm
        if self.slice_width <= 0 or self.slice_height <= 0:
            raise ValueError("Obliczone wymiary brytu są nieprawidłowe.")
        self.logger.info(LANG["log_computed_dimensions"].format(W_mm, H_mm, self.slice_width / self.mm, S_x_mm, self.slice_height / self.mm, S_y_mm, self.core_slice_width / self.mm, self.core_slice_height / self.mm, self.effective_overlap))

    def get_slice_info(self, row, col, rows, cols):
        """Generuje informacje o pojedynczym brycie."""
        x_offset = col * self.core_slice_width
        y_offset = (rows - 1 - row) * self.core_slice_height
        return {"row": row, "col": col, "rows": rows, "cols": cols, "name": f"0{chr(65 + row)}{col + 1:02d}", "offset": (x_offset, y_offset), "width": self.slice_width, "height": self.slice_height}

    # --- NOWE, ZOPTYMALIZOWANE FUNKCJE RYSUJĄCE (PyMuPDF) ---

    def _draw_registration_mark_fitz(self, shape, x, y):
        """Rysuje pojedynczy znacznik pasowania (krzyżyk) na obiekcie fitz.Shape."""
        half_w, half_h = self.mark_width / 2.0, self.mark_height / 2.0
        # Biała linia pod spodem dla "wycięcia"
        if self.reg_mark_line_width_white > 0:
            shape.draw_line(fitz.Point(x - half_w, y), fitz.Point(x + half_w, y))
            shape.draw_line(fitz.Point(x, y - half_h), fitz.Point(x, y + half_h))
            shape.finish(color=COLOR_WHITE, width=self.reg_mark_line_width_white)
        # Czarna linia na wierzchu
        shape.draw_line(fitz.Point(x - half_w, y), fitz.Point(x + half_w, y))
        shape.draw_line(fitz.Point(x, y - half_h), fitz.Point(x, y + half_h))
        shape.finish(color=COLOR_PANTONE_BLACK_C_APPROX, width=self.reg_mark_line_width_black)
        shape.commit()

    def _draw_registration_marks_custom_grid_fitz(self, page, slice_info):
        """Rysuje siatkę znaczników pasowania, używając współrzędnych strony (y=0 na górze)."""
        shape = page.new_shape()
        page_h, page_w = page.rect.height, page.rect.width
        row, col, total_rows, total_cols = slice_info["row"], slice_info["col"], slice_info["rows"], slice_info["cols"]

        half_mark_w, half_mark_h = self.mark_width / 2.0, self.mark_height / 2.0

        # Współrzędne są liczone od lewego-górnego rogu strony
        center_bottom_y = page_h - half_mark_h
        if row == 0:
             center_top_y = half_mark_h
        else:
             center_top_y = page_h - self.core_slice_height - half_mark_h

        center_left_x = half_mark_w
        center_right_x = self.core_slice_width + half_mark_w

        is_first_row, is_last_row = (row == 0), (row == total_rows - 1)
        is_first_col, is_last_col = (col == 0), (col == total_cols - 1)

        draw_tl, draw_tr, draw_bl, draw_br = False, False, False, False
        if total_rows == 1 and total_cols == 1:
            draw_tr = draw_bl = True
        elif is_first_row and not is_last_col: draw_bl = True
        elif is_last_row and not is_first_col: draw_tr = True
        elif is_first_col and not is_last_row: draw_tr = True

        if is_first_row:
            if is_first_col: draw_tr, draw_bl, draw_br = True, True, True
            elif is_last_col: draw_tl, draw_bl, draw_br = True, True, True
            else: draw_tl, draw_tr, draw_bl, draw_br = True, True, True, True
        elif is_last_row:
            if is_first_col: draw_tl, draw_tr, draw_br = True, True, True
            elif is_last_col: draw_tl, draw_tr, draw_bl = True, True, True
            else: draw_tl, draw_tr, draw_bl, draw_br = True, True, True, True
        else:
             draw_tl, draw_tr, draw_bl, draw_br = True, True, True, True

        if draw_tl: self._draw_registration_mark_fitz(shape, center_left_x, center_top_y)
        if draw_tr: self._draw_registration_mark_fitz(shape, center_right_x, center_top_y)
        if draw_bl: self._draw_registration_mark_fitz(shape, center_left_x, center_bottom_y)
        if draw_br: self._draw_registration_mark_fitz(shape, center_right_x, center_bottom_y)

    def _draw_white_stripes_fitz(self, page, slice_info):
        """Rysuje białe paski na zakładki z czarną ramką."""
        if not self.use_white_stripe or self.white_stripe <= 0: return

        shape = page.new_shape()
        page_w, page_h = page.rect.width, page.rect.height
        ws_pt = self.white_stripe * self.mm
        row, col, total_rows, total_cols = slice_info["row"], slice_info["col"], slice_info["rows"], slice_info["cols"]

        is_top_row, is_lower_row = (row == 0), (row > 0)
        is_last_col = (col == total_cols - 1)

        rects_to_draw = []
        if is_top_row and not is_last_col:
            rects_to_draw.append(fitz.Rect(page_w - ws_pt, 0, page_w, page_h))
        elif is_lower_row:
            if not is_last_col: # Kształt "L"
                rects_to_draw.append(fitz.Rect(0, 0, page_w, ws_pt))
                rects_to_draw.append(fitz.Rect(page_w - ws_pt, 0, page_w, page_h))
            else: # Tylko pasek górny
                rects_to_draw.append(fitz.Rect(0, 0, page_w, ws_pt))

        for rect in rects_to_draw:
            shape.draw_rect(rect)
        shape.finish(fill=COLOR_WHITE, color=COLOR_WHITE, width=0) # Wypełnienie

        # Rysowanie czarnej ramki
        for rect in rects_to_draw:
            shape.draw_rect(rect)
        shape.finish(color=COLOR_PANTONE_BLACK_C_APPROX, width=self.border_line_width, fill=None)

        shape.commit()

    def _draw_full_slice_border_fitz(self, page):
        shape = page.new_shape()
        shape.draw_rect(page.rect)
        shape.finish(color=COLOR_PANTONE_BLACK_C_APPROX, fill=None, width=self.border_line_width)
        shape.commit()

    def _draw_barcode_fitz(self, page, slice_info, file_path, barcode_type, layout, output_type, base_rect=None):
        """Generuje kod kreskowy jako PNG i wstawia go na stronę, z uwzględnieniem specjalnych ustawień."""
        settings = self.get_code_settings(output_type, layout, barcode_type)
        data = self.generate_barcode_data(file_path, slice_info.get('name'))
        if not data or not settings: return

        text_mode = self.config.settings.get("barcode_text_position", "side")
        show_text_in_barcode = not (barcode_type == 'code_barcode' and text_mode == 'side')

        png_bytes = barcode_qr.generate_barcode_in_memory(barcode_type, data, show_text=show_text_in_barcode)
        if not png_bytes: return

        if base_rect is None:
            base_rect = page.rect

        page_w, page_h = base_rect.width, base_rect.height
        anchor = settings.get("anchor", "bottom-left")

        if barcode_type == 'code_qr':
            img_w = img_h = settings.get('scale', 20.0) * self.mm
        else:
            img_w, img_h = settings.get('scale_x', 80.0) * self.mm, settings.get('scale_y', 10.0) * self.mm
        
        off_x, off_y = settings.get('offset_x', 0.0) * self.mm, settings.get('offset_y', 0.0) * self.mm

        x0, y0 = 0, 0
        if "right" in anchor: x0 = page_w - img_w - off_x
        else: x0 = off_x
        if "bottom" in anchor: y0 = page_h - img_h - off_y
        else: y0 = off_y
        if "center" in anchor:
            if "left" not in anchor and "right" not in anchor: x0 = (page_w - img_w) / 2 + off_x
            if "top" not in anchor and "bottom" not in anchor: y0 = (page_h - img_h) / 2 + off_y

        # Pozycjonowanie względem base_rect
        img_rect = fitz.Rect(x0, y0, x0 + img_w, y0 + img_h)
        img_rect.x0 += base_rect.x0
        img_rect.x1 += base_rect.x0
        img_rect.y0 += base_rect.y0
        img_rect.y1 += base_rect.y0

        page.insert_image(img_rect, stream=png_bytes, keep_proportion=True)

        if barcode_type == 'code_barcode' and text_mode == 'side':
            text_rect = fitz.Rect(img_rect.x1 + self.mm, img_rect.y0, page.rect.width, img_rect.y1)
            page.insert_textbox(text_rect, data, fontsize=8, fontname="helv", align=fitz.TEXT_ALIGN_LEFT)

    def _draw_trial_watermark_fitz(self, page: fitz.Page):
        page.insert_textbox(page.rect, "TRIAL VERSION", fontsize=50, color=COLOR_RED, opacity=0.15, align=fitz.TEXT_ALIGN_CENTER, rotate=45)

    def _create_complete_slice_doc(self, source_doc: fitz.Document, slice_info: dict, all_settings: dict) -> fitz.Document | None:
        """Tworzy w pamięci kompletny, pojedynczy bryt jako dokument PyMuPDF z nałożoną grafiką."""
        try:
            W, H = slice_info["width"], slice_info["height"]
            slice_doc = fitz.open()
            slice_page = slice_doc.new_page(width=W, height=H)

            offset_x, offset_y = slice_info["offset"]
            clip_rect = fitz.Rect(offset_x, self.height - offset_y - H, offset_x + W, self.height - offset_y)
            slice_page.show_pdf_page(slice_page.rect, source_doc, 0, clip=clip_rect)

            if all_settings.get("draw_slice_border", False):
                self._draw_full_slice_border_fitz(slice_page)

            self._draw_white_stripes_fitz(slice_page, slice_info)

            if all_settings.get("enable_reg_marks"):
                self._draw_registration_marks_custom_grid_fitz(slice_page, slice_info)

            if all_settings.get("slice_rotation", 0) == 180:
                slice_page.set_rotation(180)

            return slice_doc
        except Exception as e:
            self.logger.error(f"Błąd tworzenia brytu {slice_info.get('name', 'N/A')}: {e}", exc_info=True)
            return None

    def split(self, input_pdf_path, output_dir, **kwargs):
        self.logger.info(LANG["log_start_splitting_process"].format(input_pdf_path))

        source_doc, is_temp_pdf = self._load_input_as_pdf(input_pdf_path)
        if not source_doc: return False

        fragments = []
        try:
            page = source_doc.load_page(0)
            self.width, self.height = page.rect.width, page.rect.height

            self.config.settings.update(kwargs)
            rows, cols = kwargs.get('rows', 2), kwargs.get('cols', 5)

            self.calculate_dimensions(kwargs.get('overlap', 20.0), rows, cols, kwargs.get('white_stripe', 10.0), kwargs.get('add_white_to_overlap', True))

            all_slices_info = [self.get_slice_info(r, c, rows, cols) for r in range(rows) for c in range(cols)]
            ordered_slices = sorted(all_slices_info, key=lambda x: (x["row"], x["col"]))

            for slice_info in ordered_slices:
                slice_doc = self._create_complete_slice_doc(source_doc, slice_info, kwargs)
                if slice_doc:
                    fragments.append(slice_doc)

            if not fragments: raise Exception("Nie udało się wygenerować żadnych brytów.")

            output_type = kwargs.get("output_type", "output_common_sheet")
            if output_type in ["output_separate_files", "output_both"]:
                self._assemble_separate_files_fitz(ordered_slices, fragments, output_dir, input_pdf_path, kwargs)

            if output_type in ["output_common_sheet", "output_both"]:
                self._assemble_common_sheet_fitz(ordered_slices, fragments, output_dir, input_pdf_path, kwargs)

            self.logger.info(LANG["log_finished_splitting_success"].format(input_pdf_path))
            return True

        except Exception as e:
            self.logger.error(f"Krytyczny błąd w procesie cięcia: {e}", exc_info=True)
            return False
        finally:
            if source_doc: source_doc.close()
            for doc in fragments: doc.close()
            if is_temp_pdf and hasattr(source_doc, 'name') and os.path.exists(source_doc.name):
                try: os.remove(source_doc.name)
                except: pass

    def _draw_separation_lines_fitz(self, page, total_slices, layout, gap, settings):
        """Rysuje linie separacyjne na wspólnym arkuszu, używając PyMuPDF."""
        shape = page.new_shape()
        page_w, page_h = page.rect.width, page.rect.height

        # Określenie wymiaru brytu w osi układania
        slice_dim_in_layout = self.slice_height if "vertical" in layout else self.slice_width

        # Współrzędne startowe obszaru treści (od lewego-górnego rogu)
        x0, y0 = self.margin_left, self.margin_top

        line_len = page_w - self.margin_left - self.margin_right

        # Linie MIĘDZY brytami
        if settings.get("enable_separation_lines", False) and total_slices > 1:
            for i in range(1, total_slices):
                y_pos = y0 + i * slice_dim_in_layout + (i - 0.5) * gap
                shape.draw_line(fitz.Point(x0, y_pos), fitz.Point(x0 + line_len, y_pos))

        # Linie na KRAWĘDZIACH
        if settings.get("add_line_at_start", False):
            y_pos = y0 - (gap / 2)
            shape.draw_line(fitz.Point(x0, y_pos), fitz.Point(x0 + line_len, y_pos))
        if settings.get("add_line_at_end", False):
            y_pos = y0 + (total_slices * slice_dim_in_layout) + (total_slices - 1) * gap + (gap / 2)
            shape.draw_line(fitz.Point(x0, y_pos), fitz.Point(x0 + line_len, y_pos))

        # Rysowanie z efektem knockout
        gs = self.config.settings.get("graphic_settings", {})
        line_width_white = gs.get("sep_line_white_width", 0.3) * self.mm
        line_width_black = gs.get("sep_line_black_width", 0.5) * self.mm

        if line_width_white > 0:
            shape.finish(color=COLOR_WHITE, width=line_width_white)
        if line_width_black > 0:
            shape.finish(color=COLOR_PANTONE_BLACK_C_APPROX, width=line_width_black)
        shape.commit()

    def _assemble_common_sheet_fitz(self, ordered_slices, fragments, output_dir, input_pdf_path, settings):
        """Składa wspólny arkusz używając w pełni PyMuPDF."""
        layout = settings.get("layout", "layout_vertical")
        total = len(fragments)

        gs = self.config.settings.get("graphic_settings", {})
        gap = float(gs.get("slice_gap_horizontal" if "horizontal" in layout else "slice_gap_vertical", 2.0)) * self.mm

        slice_w, slice_h = self.slice_width, self.slice_height
        if "horizontal" in layout:
            slice_w, slice_h = slice_h, slice_w # Obrót

        page_w = self.margin_left + slice_w + self.margin_right
        page_h = self.margin_top + (total * slice_h) + ((total - 1) * gap) + self.margin_bottom
        
        output_doc = fitz.open()
        page = output_doc.new_page(width=page_w, height=page_h)
        
        for i, (slice_info, frag_doc) in enumerate(zip(ordered_slices, fragments)):
            y_pos = self.margin_top + i * (slice_h + gap)
            rect = fitz.Rect(self.margin_left, y_pos, self.margin_left + slice_w, y_pos + slice_h)
            
            rotation = 90 if "horizontal" in layout else 0
            page.show_pdf_page(rect, frag_doc, 0, rotate=rotation)
            
            if settings.get("enable_barcode"):
                self._draw_barcode_fitz(page, slice_info, input_pdf_path, settings.get("barcode_type"), layout, "common_sheet", base_rect=rect)

        self._draw_separation_lines_fitz(page, total, layout, gap, settings)

        if self.trial_mode:
            self._draw_trial_watermark_fitz(page)
            
        base_name = os.path.splitext(os.path.basename(input_pdf_path))[0]
        sheet_suffix = "_h_common.pdf" if "horizontal" in layout else "_v_common.pdf"
        output_path = os.path.join(output_dir, f"{base_name}{sheet_suffix}")

        output_doc.save(output_path, garbage=3, deflate=True)
        self.logger.info(f"Zapisano wspólny arkusz: {output_path}")
        self.generated_output_files.append(output_path)
        output_doc.close()

    def _assemble_separate_files_fitz(self, ordered_slices, fragments, output_dir, input_pdf_path, settings):
        """Nowa wersja składająca osobne pliki, używająca PyMuPDF do wszystkiego."""
        base_name = os.path.splitext(os.path.basename(input_pdf_path))[0]

        for i, (slice_info, frag_doc) in enumerate(zip(ordered_slices, fragments)):
            W, H = frag_doc[0].rect.width, frag_doc[0].rect.height
            page_w = self.margin_left + W + self.margin_right
            page_h = self.margin_top + H + self.margin_bottom
            
            output_doc = fitz.open()
            page = output_doc.new_page(width=page_w, height=page_h)

            target_rect = fitz.Rect(self.margin_left, self.margin_top, self.margin_left + W, self.margin_top + H)
            page.show_pdf_page(target_rect, frag_doc, 0)
            
            if settings.get("enable_barcode"):
                self._draw_barcode_fitz(page, slice_info, input_pdf_path, settings.get("barcode_type"), settings.get("layout"), "separate_files")
            
            if self.trial_mode:
                self._draw_trial_watermark_fitz(page)
            
            out_path = os.path.join(output_dir, f"{base_name}_{slice_info['name']}_v.pdf")
            output_doc.save(out_path, garbage=3, deflate=True)
            output_doc.close()
            self.logger.info(f"Zapisano plik: {out_path}")
        
        self.generated_output_files.extend([os.path.join(output_dir, f"{base_name}_{s['name']}_v.pdf") for s in ordered_slices])

    def generate_barcode_data(self, file_path, slice_name):
        try:
            filename = os.path.splitext(os.path.basename(file_path))[0]
            pattern = re.compile(r"(\d+)_(\d{1,3})_")
            match = pattern.search(filename)
            if match:
                prefix = f"{match.group(1)}_{match.group(2)}"
                return f"{prefix}_{slice_name}" if slice_name else prefix
            return filename
        except Exception as e:
            self.logger.exception(f"Błąd w generate_barcode_data: {e}")
            return str(file_path)

    def get_code_settings(self, output_type, layout, barcode_type):
        """Pobiera ustawienia (pozycja, rotacja, skala) dla kodu kreskowego/QR."""
        try:
            settings_key = output_type
            layout_key = "horizontal" if "horizontal" in layout else "vertical"

            code_settings_root = self.config.settings.get("graphic_settings", {}).get("code_settings", {})
            type_settings = code_settings_root.get(barcode_type, {})
            output_settings = type_settings.get(settings_key, {})
            layout_settings = output_settings.get(layout_key)

            if layout_settings is None:
                layout_settings = output_settings.get("default", {})
            return layout_settings.copy()
        except Exception as e:
            self.logger.error(f"Błąd pobierania ustawień kodu: {e}", exc_info=True)
            return {}
